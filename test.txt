#include <stdio.h>
// comment 1
int main() {
    int num1 = 10;
    int num2 = 5;
    int sum = 0;

    for (int i = 0; i < num1; i++) {
        if (i % 2 == 0) {
            sum += i;
        } else {
            continue; // comment 2
        }
    }

    if (sum > num2) {
        print("The sum is greater than num2.\n"); // comment 3
    } else if (sum < num2) {
        print("The sum is less than num2.\n");
    } else {
        print("The sum is equal to num2.\n");
    }

    while (num1 > 0) {
        printf("%d\n", num1);
        num1--;
    }

    int factorial = 1;
    for (int i = 1; i <= num2; i++) {
        factorial *= i;
    }

    printf("The factorial of num2 is: %d\n", factorial);

    return 0;
}
